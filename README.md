mkdir vinerproxy

cd vinerproxy

wget https://raw.githubusercontent.com/vinerProxy/vinerProxy/master/release/vinerProxyv100_linux_amd64

chmod 777 vinerProxyv100_linux_amd64

./vinerProxyv100_linux_amd64
